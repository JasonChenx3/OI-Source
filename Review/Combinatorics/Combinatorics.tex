\section{其它公式}
\begin{theorem}
    \begin{displaymath}
        \sum_{i=1}^n{i \choose k}={n+1 \choose k+1} ~k>0
    \end{displaymath}
\end{theorem}
\begin{itemize}
    \item 证明：将等式加上$(1\choose k+1)=0$，左边不断合并即可。
    \item 证明：将右式不断展开即为左式。
    \item 理解：在$n+1$个中选$k+1$个点，第$k+1$个点为哨兵，剩下$k$个在它之前
    的元素中选择。
\end{itemize}
