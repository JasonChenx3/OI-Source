\subsection{Lucas定理}
\begin{theorem}[Lucas's Theorem]
	对于非负整数$n,m$以及质数$p$，若
	\begin{eqnarray*}
		n&=&\sum_{i=0}^k{n_ip^i}\\
		m&=&\sum_{i=0}^k{m_ip^i}
	\end{eqnarray*}
	则
	\begin{displaymath}
		{n \choose m}\equiv\prod_{i=0}^k{n_i \choose m_i} \pmod{p}
	\end{displaymath}
\end{theorem}
