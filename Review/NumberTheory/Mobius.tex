\section{莫比乌斯反演}
\subsection{基本定义}
\index{M!Möbius Inversion}
\begin{theorem}
	对于数论函数$f,g$，满足$\displaystyle g(n)=\sum_{d|n}f(d)$，则有
	\begin{displaymath}
		f(n)=\sum_{d|n}\mu(d)g(\frac{n}{d})
	\end{displaymath}
\end{theorem}
\subsection{证明}
莫比乌斯反演可表示为若$g=f*1$则$f=\mu*g$。
\begin{lemma}
	$\mu*1=\epsilon$
\end{lemma}
